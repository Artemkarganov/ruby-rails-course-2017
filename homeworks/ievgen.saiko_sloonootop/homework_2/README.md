#### Класс Команда игроков 
  общие атрибуты:
    тип команды (гости или хозяева); // boolean
    имя игрока (имя, фамилия); // string
	номер (номер); // integer
	состав (основной или запас); // boolean
	капитан (да или нет); // boolean
  общие методы
    методы перемещния: (хотьба, бег, трусца, аллюр, галоп);
    инструмнты игры (игра ногами, игра головой);
	типы игры: 
      защитная игра (отбор мяча, подкат, выбивание мяча за пределы поля, перевод мяча на другую сторону, пас);
      атакующая игра (выивание мяча в сторону ворот противника, длинная передача к воротам противника, пробрасывание мяча);
	  исполнение стандартных положений (угловой, штрафной, свободнй удар, пенальти);

#### Класс Судьи
    общие атрибуты
	  имя (имя, фамилия); // string
	  тип судьи (главный или нет);  // boolean
	  вид судьи (полевой или дополнитльный);  // boolean
	общи методы:
      отслживания необходимости остановить игру (нарушение, замена, травма, время);
      фиксация нарушний правил;	  
	  
#####  Подкласс Главный судья < Судьи
	частные методы:
	  принять решение остановить игру (вне игры, аут, нарушние, замена, травма, окончание тайма или игры);
	  назначать штрафы (штрафной удар, пенальти, желтая каточка и др.);
	  удалить с поля игрока (тренера);

#### Класс Тренеры
    общие атрибуты
      тип команды (гости или хозяева); // boolean
	  имя (имя, фамилия); // string
	  тип тренера (главный или нет);  // boolean
	  объект контроля (вся команда, атака, защита, вратари);  // boolean
	общи методы:
      организация игры объкта;
      поготовка решений по корректировке игры и состава;

#####  Подкласс Главный тренер < Тренеры
	частные методы:
	  принять решение скоррктировать игру (смена схемы игры, замена игроков, подсказки игрокам);
	  ругаться с судьями;
	  
#### Класс Болельщики
    общие атрибуты
      тип команды (гости или хозяева); // boolean
	  тип болльщиков (ультрас или нет);  // boolean
	общи методы:
      поддержка своей команды (аплодисменты, свист, крики, волна);
      порицание противника (свист, звуки с негативной окраской);
      
#####  Подкласс Ультрас < Болельщики
	частные методы:
	  доп. методы поддержки (кричалки, ударные и духовы инструмнты, файеры);
	  доп. методы порицания (кричалки, ударные и духовы инструмнты, файеры, драка с ультрас противника);
